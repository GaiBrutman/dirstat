# Search Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a case-insensitive name-filter search to the tree view that hides non-matching nodes, highlights matches, and resets when cleared.

**Architecture:** `filterTree` in `utils.ts` produces a pruned `FileNode` tree containing only name-matching nodes and their ancestor directories. `App.tsx` derives `displayTree` from `result` + `searchQuery` and renders a search bar above the tree. `TreeNode` gets a `searchQuery` prop and wraps matching substrings in `<mark>`.

**Tech Stack:** React 19, TypeScript, Tailwind CSS, Vitest (new)

```typescript
import { FileNode } from "./types";

export function filterTree(node: FileNode, query: string): FileNode | null {
  const q = query.toLowerCase();
  if (!node.is_dir) {
    return node.name.toLowerCase().includes(q) ? node : null;
  }
  const nameMatches = node.name.toLowerCase().includes(q);
  const filteredChildren = node.children
    .map((child) => filterTree(child, query))
    .filter((c): c is FileNode => c !== null);
  if (nameMatches || filteredChildren.length > 0) {
    return { ...node, children: filteredChildren };
  }
  return null;
}

export function countLeaves(node: FileNode): number {
  if (!node.is_dir) return 1;
  return node.children.reduce((sum, child) => sum + countLeaves(child), 0);
}
```

```typescript
function renderName(name: string) {
  if (!searchQuery) return <span className="truncate flex-1 text-gray-100">{name}</span>;
  const idx = name.toLowerCase().indexOf(searchQuery.toLowerCase());
  if (idx === -1) return <span className="truncate flex-1 text-gray-100">{name}</span>;
  return (
    <span className="truncate flex-1 text-gray-100">
      {name.slice(0, idx)}
      <mark style={{ background: "#854d0e", color: "#fef08a", borderRadius: "2px", padding: "0 1px" }}>
        {name.slice(idx, idx + searchQuery.length)}
      </mark>
      {name.slice(idx + searchQuery.length)}
    </span>
  );
}
```
